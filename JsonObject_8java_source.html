902  void add(String name, int index) {
903  int slot = hashSlotFor(name);
904  if (index < 0xff) {
905  // increment by 1, 0 stands for empty
906  hashTable[slot] = (byte)(index + 1);

914  if ((hashTable[i] & 0xff) == index + 1) {
